  <script>
    // Render HTML safely in a <pre> block
    function escapeHtml(text) {
      return text
        .replace(/&/g, "&amp;")
        .replace(/</g, "&lt;")
        .replace(/>/g, "&gt;");
    }

    // Show plain code in preview initially
    function updatePreview() {
      const code = document.getElementById('codeInput').value;
      document.getElementById('codePreview').innerText = code;
      document.getElementById('message').textContent = '';
    }
    updatePreview();
    document.getElementById('codeInput').addEventListener('input', updatePreview);

    // Highlight search matches in preview
    function findHighlight() {
      const code = document.getElementById('codeInput').value;
      const search = document.getElementById('findInput').value;
      const preview = document.getElementById('codePreview');
      if (!search) {
        preview.innerText = code;
        document.getElementById('message').textContent = '';
        return;
      }
      // Escape HTML first, then highlight
      let htmlEscaped = escapeHtml(code);
      // Escape special regex chars:
      const safeSearch = search.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
      // Use 'g' and 'i' for global and case-insensitive
      const regex = new RegExp(safeSearch, 'gi');
      // Replace with highlight span
      const highlighted = htmlEscaped.replace(regex, function(match) {
        return `<span class="highlight">${match}</span>`;
      });
      preview.innerHTML = highlighted;
      document.getElementById('message').textContent = 'Match(es) highlighted in preview!';
    }

    // Replace all search matches in code & highlight replacements in preview
    function replaceHighlight() {
      const code = document.getElementById('codeInput').value;
      const find = document.getElementById('findInput').value;
      const replaceWith = document.getElementById('replaceInput').value;
      const preview = document.getElementById('codePreview');
      if (!find) {
        preview.innerText = code;
        document.getElementById('message').textContent = '';
        return;
      }
      // Escape HTML first
      let htmlEscaped = escapeHtml(code);
      const safeFind = find.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
      const regex = new RegExp(safeFind, 'gi');
      const highlighted = htmlEscaped.replace(regex, function() {
        return `<span class="replace-highlight">${escapeHtml(replaceWith)}</span>`;
      });
      preview.innerHTML = highlighted;
      document.getElementById('message').textContent =
        `All "${find}" replaced and highlighted in preview!`;
      // Actually update the textarea with replaced code:
      document.getElementById('codeInput').value = code.replace(regex, replaceWith);
    }

    // Copy plain code (no highlights) from textarea
    function copyCode() {
      const code = document.getElementById('codeInput').value;
      navigator.clipboard.writeText(code)
        .then(() => {
          document.getElementById('message').textContent = 'Code copied!';
        })
        .catch(() => {
          document.getElementById('message').textContent = 'Copy failed.';
        });
    }
  </script>
